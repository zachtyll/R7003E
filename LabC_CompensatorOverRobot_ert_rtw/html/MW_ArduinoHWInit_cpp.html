87void MW_Arduino_Init(void)



88{



89    //Disable watchdog for Arduino Due. watchdog



90#if (defined(_ROTH_DUE_))



91    watchdogDisable();



92#endif



93    



94    /*



95     * Initialize Serial Peripheral



96     */



97    // Always Initialize Serial0



98    Serial_begin(0, _RTT_BAUDRATE_SERIAL0_);



99    



100    /* Overrun Detection Initialization */



101#ifdef _RTT_OVERRUN_DIGITAL_PIN_



102    pinMode(_RTT_OVERRUN_DIGITAL_PIN_, OUTPUT);



103    



104    /* Init the overrun detection pin to zero or else the Pin is always in a



105     * high/tristate giving the user a wrong indication of overrun conditions */



106    digitalWrite(_RTT_OVERRUN_DIGITAL_PIN_, LOW);



107#endif



108    



109    /*



110     * Initialize Analog Reference Voltage



111     * Default - Due



112     * Default, Internal 1.1, Internal 2.56, External -  Mega 2560, Mega ADK



113     * Default, Internal, External - Other Arduino targets



114     */



115#if (defined(_ROTH_MEGA2560_))



116#if (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 1))



117    analogReference(INTERNAL1V1);



118#elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 2))



119    analogReference(INTERNAL2V56);



120#elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 3))



121    analogReference(EXTERNAL);



122#endif



123#elif defined(ARDUINO_ARM_CORTEX_M0_PLUS)



124#if (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 1))



125    analogReference(AR_INTERNAL1V0);



126#elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 2))



127    analogReference(AR_INTERNAL1V65);



128#elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 3))



129    analogReference(AR_INTERNAL2V23);



130#elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 4))



131    analogReference(AR_EXTERNAL);



132#endif



133#else



134#if (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 1))



135    analogReference(INTERNAL);



136#elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 2))



137    analogReference(EXTERNAL);



138#endif



139#endif



140    



141    /*



142     * Initialize SPI bus



143     */



144    //_RTT_SPI_SSPIN_ definition is applicable for all boards excluding Due



145#if defined(_RTT_SPI_SSPIN_)



146    MW_SPIwriteReadSetup();



147#endif



148    



149    //Initialize Modbus



150    int modStatus;



151#if defined(_RTT_MODBUS_MASTER_)



152    modStatus = ModbusRTUClient.begin(_RTT_RS485_BAUD_, _RTT_RS485_CONFIG_);



153    ModbusRTUClient.setTimeout(_RTT_MODBUS_MASTERTIMEOUT_/1000);



154#endif



155    



156#if defined(_RTT_MODBUS_SLAVE_)



157    modStatus = ModbusRTUServer.begin(_RTT_MODBUS_SLAVEID_, _RTT_RS485_BAUD_, _RTT_RS485_CONFIG_);



158#if defined(_RTT_MODBUS_CONFIGCOIL_)



159    ModbusRTUServer.configureCoils(_RTT_MODBUS_COILADDR_,_RTT_MODBUS_COILNUM_);



160#endif



161#if defined(_RTT_MODBUS_CONFIGINPUT_)



162    ModbusRTUServer.configureDiscreteInputs(_RTT_MODBUS_INPUPTADDR_,_RTT_MODBUS_INPUTNUM_);



163#endif



164#if defined(_RTT_MODBUS_CONFIGHOLDINGREG_)



165    ModbusRTUServer.configureHoldingRegisters(_RTT_MODBUS_HOLDINGREGADDR_,_RTT_MODBUS_HOLDINGREGNUM_);



166#endif



167#if defined(_RTT_MODBUS_CONFIGINPUTREG_)



168    ModbusRTUServer.configureInputRegisters(_RTT_MODBUS_INPUTREGADDR_,_RTT_MODBUS_INPUTREGNUM_);



169#endif



170#endif



171      



172/* 



173 * Initialize wifi      



174 */



175#if (defined(_RTT_WIFI_UDP_) && (_RTT_WIFI_UDP_ != 0)) || (defined(_RTT_WIFI_TCP_) && (_RTT_WIFI_TCP_ != 0))|| defined(_RTT_THINGSPEAK_WIFI_) || defined(_RTT_WEBSOCKET_PORT_)



176    /* Begin WiFi*/



177    while(configureSuccess!=WL_CONNECTED)



178    {



179        #if (defined(_RTT_DISABLE_Wifi_DHCP_) && (_RTT_DISABLE_Wifi_DHCP_ !=0))



180            WiFi.config(wifiLocalIpAddress);



181        #endif



182



183        #ifdef _RTT_WIFI_WEP



184            configureSuccess=WiFi.begin(ssid,keyIndex,key);



185        #endif //_RTT_WIFI_WEP



186



187        #ifdef _RTT_WIFI_WPA



188            configureSuccess=WiFi.begin(ssid,wpapass);



189        #endif //_RTT_WIFI_WPA



190



191        #ifdef _RTT_WIFI_NONE



192            configureSuccess=WiFi.begin(ssid);



193        #endif //_RTT_WIFI_NONE



194



195            //Wait for 10 seconds for Wifi connection to happen



196            unsigned long startMillis  = millis();



197            while(millis()- startMillis < 10000)



198            {



199            };



200    }



201



202    /* Wait for Serial0 to be up*/



203    for(trialcount=0;trialcount<5;trialcount++)



204    {



205        if(Serial)



206        {



207            break;



208        }else



209        {



210            unsigned long startMillis  = millis();



211            while(millis()- startMillis < 1000)



212            {



213            };



214        }



215    }



216



217    #if (!(defined(_RTT_WIFI_EXTMODE_) && (_RTT_WIFI_EXTMODE_ == 1)))



218        if (configureSuccess==WL_CONNECTED)



219        {



220            // If the Configuration is successful, relay back the assigned IP address.



221            IPAddress ip = WiFi.localIP();



222            Serial.print("<<< IP address :");



223            Serial.print(ip);



224            Serial.println(" >>>");            



225        }



226        else



227        {



228           // If the Configuration failed,relay back the error message.



229           Serial.println("<<< IP address :Failed to configure. >>>");



230        }



231    #endif //(!(defined(_RTT_WIFI_EXTMODE_) && (_RTT_WIFI_EXTMODE_ == 1)))  



232



233#endif /*Initialize wifi*/



234  #ifdef NANO_MOTOR_CARRIER 



235    MW_EnableNanoCarrierBatteryCharging();



236  #endif



237}



238/* Function: MW_Arduino_Terminate ================================================



239 * For Arduino MKR/Leonardo and its variants, the Virtual COM port is handled



240 * by the controller. In case the code running on the target exits main,



241 * the COM port cannot be accessed until a hard reset is performed.



242 * To over come this issue, a while loop is added to make sure that



243 * upon getting a stop command from external mode, the code running on



244 * the target stops but the code will not exit the main.



245 * This will ensure that the COM port is accessible even after the



246 * external mode has been stopped. */



247void MW_Arduino_Terminate(void){



248#if defined(ARDUINO_VIRTUAL_COM_PORT)



249    disable_rt_OneStep();



250    while(1){};     /* To continue virtual COM port functionality */



251#endif



252}

254#ifdef NANO_MOTOR_CARRIER



255void MW_EnableNanoCarrierBatteryCharging(void) {



256  Wire.begin();



257  Wire.beginTransmission(PMIC_ADDRESS);



258  Wire.write(PMIC_REG00);



259  Wire.write(0x06); // min sys voltage 3.88V + max input current 2.0 A



260  Wire.endTransmission();



261  Wire.beginTransmission(PMIC_ADDRESS);



262  Wire.write(PMIC_REG01);



263  Wire.write(0x1B); // Charge Battery + Minimum System Voltage 3.5V



264  Wire.endTransmission();



265  Wire.beginTransmission(PMIC_ADDRESS);



266  Wire.write(PMIC_REG02);



267  Wire.write(0x00); // Charge current  512 mA



268  Wire.endTransmission();



269  Wire.beginTransmission(PMIC_ADDRESS);



270  Wire.write(PMIC_REG04);



271  Wire.write(0x9E); // Charge Voltage Limit 4.128V



272  Wire.endTransmission();



273  Wire.beginTransmission(PMIC_ADDRESS);



274  Wire.write(PMIC_REG05);



275  Wire.write(0x8A); // Enable Battery Charge termination + disable watchdog



276  Wire.endTransmission();



277}
