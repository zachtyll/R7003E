112extern "C" void fillDataBuffer(const uint8_T port,const uint16_t DataSizeInBytes,char *data,uint8_T *status)



113{



114    char databuffer[2*DataSizeInBytes];    



115    (*read_func_ptr[port])(databuffer,DataSizeInBytes);



116    for(int ii=0;ii< DataSizeInBytes;ii++)



117    {



118        *data++ = databuffer[ii];



119    }



120    *status = 1;



121}

125extern "C" void MW_Serial_read(const uint8_T port,const uint16_t DataSizeInBytes, char *data,uint8_T *status)



126{



127    



128    int dataSizeinBuffer;



129    char tempdatabuffer[64];    



130    



131    



132    /* Read the size of data available in the Arduino Serial buffer */



133    dataSizeinBuffer = (*available_func_ptr[port])();



134    if (dataSizeinBuffer == DataSizeInBytes)



135    {



136        /* If the size of data in buffer == required data size, data=received data and status =1 */



137        fillDataBuffer(port,DataSizeInBytes,data,status);



138    }



139    else if ( dataSizeinBuffer >= DataSizeInBytes)



140    {



141        if (dataSizeinBuffer <= 2*DataSizeInBytes)



142        {



143            /* If the size of data in buffer > required data size and < 2*required data size, data=received data(of the rqquired size) and status =1 */



144            fillDataBuffer(port,DataSizeInBytes,data,status);



145        }



146        else



147        {



148            /* If the size of data in buffer >  2*required data size, trash data till size of data in buffer < 2*required data size.data=received data and status =1 */



149            (*read_func_ptr[port])(tempdatabuffer,(((dataSizeinBuffer/DataSizeInBytes)-1)*DataSizeInBytes));



150            fillDataBuffer(port,DataSizeInBytes,data,status);



151        }



152    }



153    else



154    {



155        /* If the size of data in buffer < required data size, data=0's  and status =0 */



156        for(int ii=0;ii< DataSizeInBytes;ii++)



157        {



158            *data++ = 0;



159        }



160        *status = 0;



161    }



162}
